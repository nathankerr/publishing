\documentclass[a5paper,article]{memoir}

% Sizes
\newcommand\sizenormal{11pt}
\newcommand\sizelarge{17.721pt} % normal * 1.611

% Setup Font and Font sizes
\usepackage{fontspec}
\setmainfont[Ligatures=TeX]{Georgia}

\renewcommand\normalsize{
	\fontsize{\sizenormal}{\sizelarge}
	\selectfont
}

%TODO: footer needs to be in rhythm
\usepackage[a5paper, twoside, includeheadfoot, width=114.6mm, height=184.6mm, inner=12.7mm, top=12.7mm, headheight=\sizenormal,headsep=\sizenormal, foot=\sizelarge]{geometry}

% Paragraph settings
%TODO: paragraphs need to maintain rhythm
\setlength\parindent{0pt}
\setlength\parskip{\sizelarge}

% pandoc starts with chapters, since we are in an article, we want it to appear like an article

%\renewcommand\chapterheadstart{\vspace{-\parskip}}
\renewcommand{\chapnamefont}{\LARGE}
\renewcommand{\chapnumfont}{\LARGE}
%\setlength\midchapskip{2pt}
\renewcommand{\chaptitlefont}{\LARGE}
\renewcommand\printchapternonum{\vphantom{\chapnamefont Chapter \chapnumfont 2}\vspace{2pt}}
\renewcommand{\printchaptertitle}[1]{%
	\chaptitlefont
	#1
}

% Section style
% Has two lines (35.442pt)
\setbeforesecskip{\sizelarge}
\setsecindent{0pt}
\setsecheadstyle{Section.\fontsize{\sizelarge}{\sizelarge}\selectfont}
\setaftersecskip{1pt}

% Subsection style
\setbeforesubsecskip{\sizenormal}
\setsubsecindent{0pt}
\setsubsecheadstyle{\bfseries}
\setaftersubsecskip{1pt}

% Subsubsection style
\setbeforesubsubsecskip{1pt}
\setsubsecindent{0pt}
\setsubsubsecheadstyle{}
\setaftersubsubsecskip{1pt}

% Paragraph style
\setbeforeparaskip{1pt}
\setparaindent{0pt}
\setparaheadstyle{\bfseries}
\setafterparaskip{-1em}

% Subparagraph style
\setbeforesubparaskip{1pt}
\setsubparaindent{\parindent}
\setsubparaheadstyle{\bfseries}
\setaftersubparaskip{-1em}
